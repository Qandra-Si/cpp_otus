24 template <class T, size_t Size>
25 struct ringbuffer_ring_t
26 {
27  std::array<T, Size> data;
28  std::atomic<unsigned> head{ 0 };
29  std::atomic<unsigned> tail{ 0 };
30 
31  [[nodiscard]] constexpr inline unsigned next(unsigned current) const noexcept
32  {
33  return (current + 1) % Size;
34  }
35 
36  [[nodiscard]] constexpr unsigned head_relaxed() const noexcept
37  {
38  return head.load(std::memory_order_relaxed);
39  }
40 
41  [[nodiscard]] constexpr unsigned tail_relaxed() const noexcept
42  {
43  return tail.load(std::memory_order_relaxed);
44  }

50  [[nodiscard]] bool push(const T& value) noexcept
51  {
52  unsigned _head = head.load(std::memory_order_relaxed);
53  unsigned next_head = next(_head);
54  if (next_head == tail.load(std::memory_order_acquire))
55  return false;
56  data[_head] = value;
57  head.store(next_head, std::memory_order_release);
58  return true;
59  }
60 
65  [[nodiscard]] bool pop(T& value) noexcept
66  {
67  unsigned _tail = tail.load(std::memory_order_relaxed);
68  if (_tail == head.load(std::memory_order_acquire))
69  return false;
70  value = std::move(data[_tail]);
71  tail.store(next(_tail), std::memory_order_release);
72  return true;
73  }
74 };
