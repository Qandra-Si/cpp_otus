55  // Allocate storage for n values of T
56  T *allocate(size_type n, const void* /*[[maybe_unused]] y*/ = 0)
57  {
58  // C++ [allocator.requirements.general] #5 Note C
59  // If n == 0, the return value is unspecified.
60  if (n == 0) return nullptr;
61  // Memory is allocated for an array of n T and such an object is created but
62  // array elements are not constructed.
63  auto p = heap->malloc(n * sizeof(T));
64  if (!p)
65  throw std::bad_alloc();
66  return reinterpret_cast<T*>(p);
67  }

71  {
